## Surface preview

```ts
import { Sovseal } from "@sovseal/sdk";

const memory = new Sovseal({
  // optional — defaults to ~/.sovseal/config.json
  projectToken: process.env.SOVSEAL_PROJECT_TOKEN,
  // optional — set null to disable replication entirely (local-only mode)
  replicationUrl: "https://api.sovseal.com/v2/agent-state",
});

await memory.store("user.preferences", { theme: "dark" });
const hits = await memory.recall("user theme preference", { topK: 5 });
```
